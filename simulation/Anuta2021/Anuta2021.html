"""Anuta2021"""

from typing import Dict

from sbmlsim.data import DataSet, load_pkdb_dataframe
from sbmlsim.fit import FitMapping, FitData
from sbmlutils.console import console

from pkdb_models.models.captopril.experiments.base_experiment import (
    CaptoprilSimulationExperiment,
)
from pkdb_models.models.captopril.experiments.metadata import Tissue, Route, Dosing, Health, Fasting, \
    CaptoprilMappingMetaData, PKPDData
from sbmlsim.plot import Axis, Figure
from sbmlsim.simulation import Timecourse, TimecourseSim

from pkdb_models.models.captopril.helpers import run_experiments


class Anuta2021(CaptoprilSimulationExperiment):
    """Simulation experiment for Anuta2021.
    Pharmacokinetics of captopril in bioequivalence studies.
    - no information about food
    """

    routes = "PO"
    doses = []  # mg
    yids = []
    clabels = []

    colors = {
        "50": "#3C33FF",
        "100": "black",
    }

    marker = {
        "exp1": "s",
        "exp2": "o",
        "exp3": "x",
        "exp4": "^",
        "exp5": "*",
        "exp6": "D"
    }

    def datasets(self) -> Dict[str, DataSet]:
        dsets = {}
        self.doses = []
        self.yids = []
        self.clabels = []
        for fig_id in ["Fig2"]:
            df = load_pkdb_dataframe(f"{self.sid}_{fig_id}", data_path=self.data_path)
            for label, df_label in df.groupby("label"):
                dset = DataSet.from_df(df_label, self.ureg)
                if label.__contains__("_T_") or label.__contains__("_R_"):
                    dset.unit_conversion("mean", 1 / self.Mr.cap)
                    if "50" in label:
                        self.doses.append(50)
                    else:
                        self.doses.append(100)
                    self.yids.append("[Cve_cap]")
                    self.clabels.append(label)
                    dsets[f"{label}"] = dset
        # console.print(dsets.keys())
        # console.print(dsets)
        return dsets

    def simulations(self) -> Dict[str, TimecourseSim]:
        Q_ = self.Q_
        tcsims = {}
        for k, dose in enumerate(self.doses):
            tcsims[f"cap_{self.routes}_{dose}"] = TimecourseSim(
                Timecourse(
                    start=0,
                    end=10 * 60,  # minutes
                    steps=400,
                    changes={
                        **self.default_changes(),
                        f"{self.routes}DOSE_cap": Q_(dose, "mg"),
                    },
                )
            )
        return tcsims

    def fit_mappings(self) -> Dict[str, FitMapping]:
        mappings = {}
        for kr, dose in enumerate(self.doses):
            if "R_" or "T_" in self.clabels[kr]:
                mappings[
                    f"fm_cap{self.routes}{dose}_{kr}_cap_plasma_acute_nrfasting_all"
                ] = FitMapping(
                    self,
                    reference=FitData(
                        self,
                        dataset=self.clabels[kr],
                        xid="time",
                        yid="mean",
                        yid_sd="mean_sd",
                        count="count",
                    ),
                    observable=FitData(
                        self,
                        task=f"task_cap_{self.routes}_{dose}",
                        xid="time",
                        yid=self.yids[kr],
                    ),
                    metadata=CaptoprilMappingMetaData(
                        tissue=Tissue.PLASMA,
                        route=Route.PO,
                        dosing=Dosing.SINGLE,
                        health=Health.HEALTHY,
                        fasting=Fasting.NR,
                        data=PKPDData.PK,
                    ),
                )
        # console.print(mappings)
        return mappings

    def figures(self) -> Dict[str, Figure]:
        fig = Figure(
            experiment=self,
            sid="Figure 2",
            num_rows=1,
            num_cols=1,
            name=f"{self.__class__.__name__}",
            height=self.panel_height,
            width=self.panel_width,
        )
        doses_unique = [100, 50]
        label_unique = "[Cve_cap]"
        plots = fig.create_plots(
            xaxis=Axis(self.labels["time"], unit=self.units["time"]), legend=True
        )
        for kr, dose in enumerate(doses_unique):
            plots[0].set_yaxis(self.labels[label_unique], unit=self.units[label_unique])
            plots[0].add_data(
                task=f"task_cap_{self.routes}_{dose}",
                xid="time",
                yid=self.yids[0],
                label=f"sim: {dose}mg {self.routes}",
                color=self.colors[f"{dose}"],
            )
        for kr, dose in enumerate(self.doses):
            if "R_cap" or "T_cap" in self.clabels[kr]:
                # dataset
                plots[0].add_data(
                    dataset=self.clabels[kr],
                    xid="time",
                    yid="mean",
                    #yid_sd="mean_sd",
                    count="count",
                    label=f"exp{kr+1}: {dose}mg {self.routes}",
                    color=self.colors[f"{dose}"],
                    marker=self.marker[f"exp{kr+1}"],
                )
        return {
            fig.sid: fig,
        }


if __name__ == "__main__":
    run_experiments(Anuta2021, output_dir=Anuta2021.__name__)
