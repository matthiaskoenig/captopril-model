from typing import Dict

from sbmlsim.data import DataSet, load_pkdb_dataframe
from sbmlsim.fit import FitMapping, FitData
from sbmlutils.console import console

from pkdb_models.models.captopril.experiments.base_experiment import (
    CaptoprilSimulationExperiment,
)
from pkdb_models.models.captopril.experiments.metadata import Tissue, Route, Dosing, Health, Fasting, \
    CaptoprilMappingMetaData, PKPDData
from sbmlsim.plot import Axis, Figure
from sbmlsim.simulation import Timecourse, TimecourseSim

from pkdb_models.models.captopril.helpers import run_experiments


class Chik2014(CaptoprilSimulationExperiment):
    """Simulation experiment of Chik2014.
    A Bioequivalence Comparison of Two Captopril Formulations (25 mg Tablets).
    - single dose
    """

    colors = {
        "control": CaptoprilSimulationExperiment.renal_colors["Control"],
    }
    clabels = []
    drugs = []

    marker = {
        "exp1": "s",
        "exp2": "o",
        "exp3": "x",
        "exp4": "^",
        "exp5": "*",
        "exp6": "D"
    }

    def datasets(self) -> Dict[str, DataSet]:
        dsets = {}
        self.clabels = []
        for fig_id in ["Fig1"]:
            df = load_pkdb_dataframe(f"{self.sid}_{fig_id}", data_path=self.data_path)

            for label, df_label in df.groupby("label"):
                dset = DataSet.from_df(df_label, self.ureg)

                dset.unit_conversion("mean", 1 / self.Mr.cap)
                dsets[f"{label}"] = dset
                self.clabels.append(label)

        # console.print(dsets.keys())
        return dsets

    def simulations(self) -> Dict[str, TimecourseSim]:
        Q_ = self.Q_
        tcsims = {}
        tcsims[f"po_cap25"] = TimecourseSim(
            [
                Timecourse(
                    start=0,
                    end=12 * 60,  # [min]
                    steps=500,
                    changes={
                        **self.default_changes(),
                        "PODOSE_cap": Q_(25, "mg"),
                        #"BMI": Q_(22.15, "kg / m ** 2"),
                        "HEIGHT": Q_(165.3, "cm"),
                        "BW": Q_(60.6, "kg"),
                    },
                )
            ]
        )

        return tcsims

    def fit_mappings(self) -> Dict[str, FitMapping]:

        mappings = {}
        for label in self.clabels:
            if label.startswith("CAPR25_"):
                intervention = "capr25"
            else:
                intervention = "cap25"
            mappings[f"fm_{intervention}_cap_serum_acute_nonfasting_all"] = FitMapping(
                self,
                reference=FitData(
                    self,
                    dataset=label,
                    xid="time",
                    yid="mean",
                    yid_sd="mean_sd",
                    count="count",
                ),
                observable=FitData(
                    self,
                    task="task_po_cap25",
                    xid="time",
                    yid=f"[Cve_cap]",
                ),
                metadata=CaptoprilMappingMetaData(
                    tissue=Tissue.PLASMA,
                    route=Route.PO,
                    dosing=Dosing.SINGLE,
                    health=Health.HEALTHY,
                    fasting=Fasting.FASTING,
                    data=PKPDData.PK,
                ),
            )

        # console.print(mappings)
        return mappings

    def figures(self) -> Dict[str, Figure]:

        fig = Figure(
            experiment=self,
            sid="Figure 1",
            num_rows=1,
            num_cols=1,
            name=f"{self.__class__.__name__}",
            height=self.panel_height,
            width=self.panel_width,
        )
        plots = fig.create_plots(
            xaxis=Axis(self.labels["time"], unit=self.units["time"]), legend=True
        )
        plots[0].set_yaxis(self.labels["[Cve_cap]"], unit=self.units["[Cve_cap]"])

        # simulation
        plots[0].add_data(
            task=f"task_po_cap25",
            xid="time",
            yid=f"[Cve_cap]",
            label="sim: 25mg PO",
            color=self.colors["control"],
        )
        # data
        for kl, label in enumerate(self.clabels):
            plots[0].add_data(
                dataset=label,
                xid="time",
                yid="mean",
                # yid_sd="mean_sd",
                count="count",
                label=f"exp{kl+1}: 25mg PO",
                color=self.colors["control"],
                marker=self.marker[f"exp{kl+1}"],
            )

        return {
            fig.sid: fig,
        }


if __name__ == "__main__":
    run_experiments(Chik2014, output_dir=Chik2014.__name__)
