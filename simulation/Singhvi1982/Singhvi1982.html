"""Singhvi1982 experiment"""

from typing import Dict

from sbmlsim.data import DataSet, load_pkdb_dataframe
from sbmlsim.fit import FitMapping, FitData
from sbmlutils.console import console

from pkdb_models.models.captopril.experiments.base_experiment import (
    CaptoprilSimulationExperiment,
)
from pkdb_models.models.captopril.experiments.metadata import Tissue, Route, Dosing, Health, Fasting, \
    CaptoprilMappingMetaData, PKPDData
from sbmlsim.plot import Axis, Figure
from sbmlsim.simulation import Timecourse, TimecourseSim

from pkdb_models.models.captopril.helpers import run_experiments


class Singhvi1982(CaptoprilSimulationExperiment):
    """Simulation experiment for Singhvi1982.
    Determination of an effect of food on the bioavailability of captopril in healthy subjects.
    -8 hr fasting, light breakfast for the second gruop, no food for 4 hr
    """

    route = "PO"
    dose = 100  # mg
    yids = []
    clabels = []
    states = []
    substances = []
    tissues = []

    states_unique = [
        "nonfasting",
        "fasted"
    ]


    colors = {
        "nonfasting": CaptoprilSimulationExperiment.fasting_colors["nonfasting"],
        "fasted": CaptoprilSimulationExperiment.fasting_colors["fasted"],
        "control": CaptoprilSimulationExperiment.renal_colors["Control"]
    }

    def datasets(self) -> Dict[str, DataSet]:
        dsets = {}
        Q_ = self.Q_
        self.clabels = []
        self.tissues = []
        self.yids = []
        self.substances = []
        self.states = []
        recovery_to_cumamount = Q_(self.dose, "mg") / Q_(100, "percent")
        for fig_id in ["Fig1", "Tab3", "Tab2"]:
            df = load_pkdb_dataframe(f"{self.sid}_{fig_id}", data_path=self.data_path)
            for label, df_label in df.groupby("label"):
                dset = DataSet.from_df(df_label, self.ureg)
                if "_captopril_" in label:
                    if "blood" in label:
                        dset.unit_conversion(
                            "mean", 1 / self.Mr.cap / self.BPRatios.cap
                        )
                        self.yids.append("[Cve_cap]")
                        self.tissues.append("plasma")
                    else:
                        dset.unit_conversion(
                            "mean", 1 / self.Mr.cap * recovery_to_cumamount
                        )
                        self.yids.append("Aurine_cap")
                        self.tissues.append("urine")
                    self.substances.append("cap")
                elif "_captopril+" in label and fig_id == "Tab2":

                    if "urine" in label:
                        self.yids.append("Aurine_capall")
                        self.tissues.append("urine")
                        dset.unit_conversion(
                            "mean", 1 / self.Mr.capall * recovery_to_cumamount
                        )
                        self.substances.append("capall")
                    else:
                        self.yids.append("Afeces_captot")
                        self.tissues.append("feces")
                        dset.unit_conversion(
                            "mean", 1 / self.Mr.captot * recovery_to_cumamount
                        )
                        self.substances.append("captot")
                else:
                    continue
                if "nonfasting" in label:
                    self.states.append("nonfasting")
                else:
                    self.states.append("fasted")
                self.clabels.append(label)
                dsets[f"{label}"] = dset
        # console.print(dsets.keys())
        # console.print(dsets)
        return dsets

    def simulations(self) -> Dict[str, TimecourseSim]:
        Q_ = self.Q_
        tcsims = {}
        for state_unique in self.states_unique:
            tcsims[f"cap_{self.route}_{self.dose}_{state_unique}"] = TimecourseSim(
                Timecourse(
                    start=0,
                    end=75 * 60,  # minutes
                    steps=1000,
                    changes={
                        **self.default_changes(),
                        "BW": Q_(78.2, "kg"),
                        "HEIGHT": Q_(181.3, "cm"),
                        f"{self.route}DOSE_cap": Q_(self.dose, "mg"),
                        f"GU__F_cap_abs": Q_(self.fasting_map[state_unique], "dimensionless"),
                    },
                )
            )

        return tcsims

    def fit_mappings(self) -> Dict[str, FitMapping]:
        mappings = {}
        for kr, dset_id in enumerate(self.clabels):
            tissue = self.tissues[kr]
            substance = self.substances[kr]
            state = self.states[kr]
            mappings[f"fm_cap{self.route}{self.dose}_{substance}_{tissue}_{state}"] = (
                FitMapping(
                    self,
                    reference=FitData(
                        self,
                        dataset=dset_id,
                        xid="time",
                        yid="mean",
                        yid_sd="mean_sd",
                        count="count",
                    ),
                    observable=FitData(
                        self,
                        task=f"task_cap_{self.route}_{self.dose}_{state}",
                        xid="time",
                        yid=self.yids[kr],
                    ),
                    metadata=CaptoprilMappingMetaData(
                        tissue=(
                            Tissue.PLASMA
                            if "plasma" in tissue
                            else Tissue.URINE if "urine" in tissue else Tissue.FECES
                        ),
                        route=Route.PO,
                        dosing=Dosing.SINGLE,
                        health=Health.HEALTHY,
                        fasting=(
                            Fasting.NONFASTING if state == "nonfasting" else Fasting.FASTING
                        ),
                        data=PKPDData.PK
                    ),
                )
            )
        return mappings

    def figures(self) -> Dict[str, Figure]:
        fig1 = Figure(
            experiment=self,
            sid="Figure 1",
            num_rows=1,
            num_cols=1,
            name=f"{self.__class__.__name__}",
            height=self.panel_height,
            width=self.panel_width,
        )
        plots1 = fig1.create_plots(
            xaxis=Axis(self.labels["time"], unit=self.units["time"]), legend=True
        )
        fig2 = Figure(
            experiment=self,
            sid="Tabs 1,3",
            num_rows=1,
            num_cols=3,
            name=f"{self.__class__.__name__}",
            height=self.panel_height,
            width=self.panel_width * 3,
        )
        plots2 = fig2.create_plots(
            xaxis=Axis(self.labels["time"], unit=self.units["time"]), legend=True
        )

        for ky, yid in enumerate(["[Cve_cap]", "Aurine_cap", "Aurine_capall", "Afeces_captot"]):
            if yid == "[Cve_cap]":
                plots = plots1
                plots[ky].set_yaxis(self.labels[yid], unit=self.units[yid])
                plots[ky].xaxis.max = 9
                plots[ky].xaxis.min = -0.1
                plots[ky].yaxis.max = 8
            else:
                plots = plots2
                ky -= 1
                plots[ky].set_yaxis(self.labels[yid], unit=self.units[yid])
                plots[ky].xaxis.max = 75
                plots[ky].xaxis.min = 0
                plots[ky].yaxis.max = 301

            for state_unique in self.states_unique:
                plots[ky].add_data(
                    task=f"task_cap_{self.route}_{self.dose}_{state_unique}",
                    xid="time",
                    yid=yid,
                    label=f"{state_unique} sim: {self.dose}mg {self.route}",
                    color=self.colors[state_unique],
                )

            for kr, dset_id in enumerate(self.clabels):
                if self.yids[kr] == yid:
                    plots[ky].add_data(
                        dataset=dset_id,
                        xid="time",
                        yid="mean",
                        yid_sd="mean_sd",
                        count="count",
                        label=f"{self.states[kr]} exp: {self.dose}mg {self.route}",
                        color=self.colors[f"{self.states[kr]}"],
                    )

        return {
            fig1.sid: fig1,
            fig2.sid: fig2,
        }


if __name__ == "__main__":
    run_experiments(Singhvi1982, output_dir=Singhvi1982.__name__)
