"""Jankowski1995"""

from typing import Dict

from sbmlsim.data import DataSet, load_pkdb_dataframe
from sbmlsim.fit import FitMapping, FitData
from sbmlutils.console import console

from pkdb_models.models.captopril.experiments.base_experiment import (
    CaptoprilSimulationExperiment,
)
from pkdb_models.models.captopril.experiments.metadata import Tissue, Route, Dosing, Health, Fasting, \
    CaptoprilMappingMetaData, PKPDData
from sbmlsim.plot import Axis, Figure
from sbmlsim.simulation import Timecourse, TimecourseSim

from pkdb_models.models.captopril.helpers import run_experiments


class Jankowski1995(CaptoprilSimulationExperiment):
    """Simulation experiment for Jankowski1995.
    The study describes a method for determination of unchanged captopril,
    an angiotensin-converting enzyme inhibitor, in human plasma.
    - fasting overnight, meal after 3hr
    """

    route = "PO"
    doses = []  # mg
    yids = []
    clabels = []

    colors = {
        "25": CaptoprilSimulationExperiment.doses_colors["50mg"],
        "50": CaptoprilSimulationExperiment.doses_colors["100mg"],
    }

    def datasets(self) -> Dict[str, DataSet]:
        dsets = {}
        self.doses = []
        self.clabels = []
        self.yids = []
        for fig_id in ["Fig2"]:
            df = load_pkdb_dataframe(f"{self.sid}_{fig_id}", data_path=self.data_path)
            for label, df_label in df.groupby("label"):
                dset = DataSet.from_df(df_label, self.ureg)
                dset.unit_conversion("mean", 1 / self.Mr.cap)
                if "25mg" in label:
                    self.doses.append(25)
                else:
                    self.doses.append(50)
                self.yids.append("[Cve_cap]")
                self.clabels.append(label)
                dsets[f"{label}"] = dset
        # console.print(dsets.keys())
        # console.print(dsets)
        return dsets

    def simulations(self) -> Dict[str, TimecourseSim]:
        Q_ = self.Q_
        tcsims = {}
        for k, dose in enumerate(self.doses):
            tcsims[f"cap_{self.route}_{dose}"] = TimecourseSim(
                Timecourse(
                    start=0,
                    end=7 * 60,  # minutes
                    steps=400,
                    changes={
                        **self.default_changes(),
                        f"{self.route}DOSE_cap": Q_(dose, "mg"),
                    },
                )
            )
        return tcsims

    def fit_mappings(self) -> Dict[str, FitMapping]:
        mappings = {}
        for kr, dose in enumerate(self.doses):
            mappings[f"fm_cap{self.route}{dose}_cap_plasma_acute_fasting_all"] = (
                FitMapping(
                    self,
                    reference=FitData(
                        self,
                        dataset=self.clabels[kr],
                        xid="time",
                        yid="mean",
                        yid_sd="mean_sd",
                        count="count",
                    ),
                    observable=FitData(
                        self,
                        task=f"task_cap_{self.route}_{dose}",
                        xid="time",
                        yid=self.yids[kr],
                    ),
                    metadata=CaptoprilMappingMetaData(
                        tissue=Tissue.PLASMA,
                        route=Route.PO,
                        dosing=Dosing.SINGLE,
                        health=Health.HEALTHY,
                        fasting=Fasting.FASTING,
                        data=PKPDData.PK
                    ),
                )
            )
        # console.print(mappings)
        return mappings

    def figures(self) -> Dict[str, Figure]:
        fig = Figure(
            experiment=self,
            sid="Figure 2",
            num_rows=1,
            num_cols=1,
            name=f"{self.__class__.__name__}",
            height=self.panel_height,
            width=self.panel_width,
        )
        plots = fig.create_plots(
            xaxis=Axis(self.labels["time"], unit=self.units["time"]), legend=True
        )
        for kr, dose in enumerate(self.doses):
            yid = self.yids[kr]
            dset_id = self.clabels[kr]
            plots[0].set_yaxis(self.labels[yid], unit=self.units[yid])
            # simulation
            plots[0].add_data(
                task=f"task_cap_{self.route}_{dose}",
                xid="time",
                yid=self.yids[kr],
                label=f"sim: {dose}mg PO",
                color=self.colors[f"{dose}"],
            )
            # dataset
            plots[0].add_data(
                dataset=dset_id,
                xid="time",
                yid="mean",
                yid_sd="mean_sd",
                count="count",
                label=f"exp: {dose}mg PO",
                color=self.colors[f"{dose}"],
            )
        return {
            fig.sid: fig,
        }


if __name__ == "__main__":
    run_experiments(Jankowski1995, output_dir=Jankowski1995.__name__)
