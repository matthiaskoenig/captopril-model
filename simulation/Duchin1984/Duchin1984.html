"""Duchin1984 experiment"""

from typing import Dict

from sbmlsim.data import DataSet, load_pkdb_dataframe
from sbmlsim.fit import FitMapping, FitData
from sbmlutils.console import console
from sympy.strategies.core import switch

from pkdb_models.models.captopril.experiments.base_experiment import (
    CaptoprilSimulationExperiment,
)
from pkdb_models.models.captopril.experiments.metadata import Tissue, Route, Dosing, Health, Fasting, \
    CaptoprilMappingMetaData, PKPDData
from sbmlsim.plot import Axis, Figure
from sbmlsim.simulation import Timecourse, TimecourseSim

from pkdb_models.models.captopril.helpers import run_experiments


class Duchin1984(CaptoprilSimulationExperiment):
    """Simulation experiment for Duchin1984.
    Elimination kinetics of captopril in patients with renal failure.
    - fasting 4hr before and 2 hr after
    - sd checked
    """

    route = "PO"
    dose = 100  # mg
    yids = []
    clabels = []
    groups = []
    tissues = []
    substances = []

    colors = {
        "MoRI": CaptoprilSimulationExperiment.renal_colors["Moderate renal impairment"],
        "MiRI": CaptoprilSimulationExperiment.renal_colors["Mild renal impairment"],
        "SRI": CaptoprilSimulationExperiment.renal_colors["Severe renal impairment"],
        "INTD": CaptoprilSimulationExperiment.renal_colors["Interdialysis"],
    }

    renal_function = {
        "MoRI": 61 / 100,
        "MiRI": 32 / 100,
        "SRI": 10 / 100,
        "INTD": 0 / 100,
    }
    
    bodyweight = {
        "MoRI": 77.9,
        "MiRI": 70.2,
        "SRI": 67.3,
        "INTD": 70,
    }
    groups_unique = list(renal_function.keys())

    def datasets(self) -> Dict[str, DataSet]:
        dsets = {}
        Q_ = self.Q_
        recovery_to_cumamount = Q_(self.dose, "mg") / Q_(100, "percent")

        for fig_id in ["Fig1", "Tab2Mean", "Tab3Mean"]:
            df = load_pkdb_dataframe(f"{self.sid}_{fig_id}", data_path=self.data_path)
            for label, df_label in df.groupby("label", sort=False):
                dset = DataSet.from_df(df_label, self.ureg)

                if fig_id == "Tab3Mean":
                    if "recovery" in label:
                        dset.unit_conversion(
                            "mean", 1 / self.Mr.cap * recovery_to_cumamount
                        )
                        self.yids.append("Aurine_cap")
                        self.tissues.append("urine")
                        self.substances.append("cap")
                    else:
                        continue
                elif fig_id in ["Fig1", "Tab2Mean"]:
                    if "blood" in label:
                        dset.unit_conversion(
                            "mean", 1 / self.Mr.cap / self.BPRatios.cap
                        )
                        self.yids.append("[Cve_cap]")
                        self.tissues.append("plasma")
                        self.substances.append("cap")
                    elif "urine" in label:
                        dset.unit_conversion(
                            "mean", 1 / self.Mr.capall * recovery_to_cumamount
                        )
                        self.yids.append("Aurine_capall")
                        self.tissues.append("urine")
                        self.substances.append("capall")
                    elif "feces" in label:
                        dset.unit_conversion(
                            "mean", 1 / self.Mr.captot * recovery_to_cumamount
                        )
                        self.yids.append("Afeces_captot")
                        self.tissues.append("feces")
                        self.substances.append("captot")
                    else:
                        continue

                if "MiRI" in label:
                    self.groups.append("MiRI")
                elif "MoRI" in label:
                    self.groups.append("MoRI")
                elif "SRI" in label:
                    self.groups.append("SRI")
                else:
                    self.groups.append("INTD")

                self.clabels.append(label)
                dsets[f"{label}"] = dset
        # console.print(dsets.keys())
        # console.print(dsets)

        return dsets

    def simulations(self) -> Dict[str, TimecourseSim]:
        Q_ = self.Q_
        tcsims = {}
        for group in self.groups_unique:
            tcsims[f"cap_{self.route}_{self.dose}_{group}"] = TimecourseSim(
                Timecourse(
                    start=0,
                    end=122 * 60,  # minutes
                    steps=1000,
                    changes={
                        **self.default_changes(),
                        "BW": Q_(self.bodyweight[group], "kg"),
                        f"{self.route}DOSE_cap": Q_(self.dose, "mg"),
                        "KI__f_renal_function": Q_(
                            self.renal_function[group], "dimensionless"
                        ),
                    },
                )
            )
        return tcsims

    def fit_mappings(self) -> Dict[str, FitMapping]:
        mappings = {}

        for kr, dset_id in enumerate(self.clabels):
            group = self.groups[kr]
            tissue = self.tissues[kr]
            substance = self.substances[kr]
            mappings[
                f"fm_cap{self.route}_{self.dose}_{substance}_{tissue}_acute_fasting_{group}"
            ] = FitMapping(
                self,
                reference=FitData(
                    self,
                    dataset=dset_id,
                    xid="time",
                    yid="mean",
                    yid_sd="mean_sd",
                    count="count",
                ),
                observable=FitData(
                    self,
                    task=f"task_cap_{self.route}_{self.dose}_{group}",
                    xid="time",
                    yid=self.yids[kr],
                ),
                metadata=CaptoprilMappingMetaData(
                    tissue=(
                        Tissue.PLASMA
                        if "blood" in dset_id
                        else Tissue.URINE if "urine" in dset_id else Tissue.FECES
                    ),
                    route=Route.PO,
                    dosing=Dosing.SINGLE,
                    health=Health.RENAL_IMPAIRMENT,
                    fasting=Fasting.FASTING,
                    data=PKPDData.PK
                ),
            )

        # console.print(mappings)
        return mappings

    def figures(self) -> Dict[str, Figure]:
        fig1 = Figure(
            experiment=self,
            sid="Figure 2,3",
            num_rows=1,
            num_cols=1,
            name=f"{self.__class__.__name__}",
            height=self.panel_height,
            width=self.panel_width,
        )
        plots = fig1.create_plots(
            xaxis=Axis(self.labels["time"], unit=self.units["time"]), legend=True
        )
        plots[0].set_yaxis(self.labels["[Cve_cap]"], unit=self.units["[Cve_cap]"])
        plots[0].xaxis.max = 25
        plots[0].xaxis.min = 0
        for kr, dset_id in enumerate(self.clabels):
            yid = self.yids[kr]
            if yid == "[Cve_cap]":
                # simulation
                plots[0].add_data(
                    task=f"task_cap_{self.route}_{self.dose}_{self.groups[kr]}",
                    xid="time",
                    yid=yid,
                    label=f"{self.groups[kr]} sim: {self.dose}mg PO",
                    color=self.colors[f"{self.groups[kr]}"],
                )
                # dataset
                # dataset
                plots[0].add_data(
                    dataset=dset_id,
                    xid="time",
                    yid="mean",
                    yid_sd="mean_sd",
                    count="count",
                    label=f"{self.groups[kr]} exp: {self.dose}mg PO",
                    color=self.colors[self.groups[kr]],
                )

        fig2 = Figure(
            experiment=self,
            sid="Tabs 2,3",
            num_rows=1,
            num_cols=3,
            name=f"{self.__class__.__name__}",
            height=self.panel_height,
            width=self.panel_width * 3,
        )
        plots = fig2.create_plots(
            xaxis=Axis(self.labels["time"], unit=self.units["time"]), legend=True
        )

        for kyu, yid_unique in enumerate(["Aurine_cap", "Aurine_capall", "Afeces_captot"]):
            plots[kyu].set_yaxis(self.labels[yid_unique], unit=self.units[yid_unique])
            for kr, dset_id in enumerate(self.clabels):
                yid = self.yids[kr]
                if yid == yid_unique:
                    group = self.groups[kr]
                    # simulation
                    plots[kyu].add_data(
                        task=f"task_cap_{self.route}_{self.dose}_{group}",
                        xid="time",
                        yid=yid,
                        label=f"{group} sim: {self.dose}mg PO",
                        color=self.colors[group],
                    )
                    # dataset
                    plots[kyu].add_data(
                        dataset=dset_id,
                        xid="time",
                        yid="mean",
                        yid_sd="mean_sd",
                        count="count",
                        label=f"{group} exp: {self.dose}mg PO",
                        color=self.colors[group],
                    )

        return {
            fig1.sid: fig1,
            fig2.sid: fig2,
        }


if __name__ == "__main__":
    run_experiments(Duchin1984, output_dir=Duchin1984.__name__)
