"""Kok1997"""

from typing import Dict

from sbmlsim.data import DataSet, load_pkdb_dataframe
from sbmlsim.fit import FitMapping, FitData
from sbmlutils.console import console

from pkdb_models.models.captopril.experiments.base_experiment import (
    CaptoprilSimulationExperiment,
)
from pkdb_models.models.captopril.experiments.metadata import Tissue, Route, Dosing, Health, Fasting, \
    CaptoprilMappingMetaData, PKPDData
from sbmlsim.plot import Axis, Figure
from sbmlsim.simulation import Timecourse, TimecourseSim

from pkdb_models.models.captopril.helpers import run_experiments


class Kok1997(CaptoprilSimulationExperiment):
    """Simulation experiment for Kok1997.
    Comparison of two sensitive high-performance liquid chromatographic methods for captopril pharmacokinetics measurement.
    Only one individual.
    - no information about food
    """

    routes = "PO"
    dose = "25"  # mg
    yids = []
    clabels = []
    substances = []

    marker = {
        "exp1": "v",
        "exp2": "o",
    }

    colors = {"control": CaptoprilSimulationExperiment.renal_colors["Control"]}

    def datasets(self) -> Dict[str, DataSet]:
        dsets = {}
        self.clabels = []
        self.yids = []
        self.substances = []
        for fig_id in ["Fig9"]:
            df = load_pkdb_dataframe(f"{self.sid}_{fig_id}", data_path=self.data_path)
            for label, df_label in df.groupby("label"):
                dset = DataSet.from_df(df_label, self.ureg)
                if "total captopril" in label:
                    dset.unit_conversion("value", 1 / self.Mr.captot)
                    self.yids.append("[Cve_captot]")
                    self.substances.append("captot")
                else:
                    dset.unit_conversion("value", 1 / self.Mr.cap)
                    self.yids.append("[Cve_cap]")
                    self.substances.append("cap")
                self.clabels.append(label)
                dsets[f"{label}"] = dset
        # console.print(dsets.keys())
        # console.print(dsets)
        return dsets

    def simulations(self) -> Dict[str, TimecourseSim]:
        Q_ = self.Q_
        tcsims = {}
        tcsims[f"cap_{self.routes}_{self.dose}"] = TimecourseSim(
            Timecourse(
                start=0,
                end=26 * 60,  # minutes
                steps=400,
                changes={
                    **self.default_changes(),
                    f"{self.routes}DOSE_cap": Q_(self.dose, "mg"),
                },
            )
        )
        return tcsims

    def fit_mappings(self) -> Dict[str, FitMapping]:
        mappings = {}
        for kr, dset_id in enumerate(self.clabels):
            substance = self.substances[kr]
            subjects = dset_id.split("_")[-1]
            mappings[
                f"fm_cap{self.routes}{self.dose}_{substance}_plasma_acute_fasting_{subjects}"
            ] = FitMapping(
                self,
                reference=FitData(
                    self,
                    dataset=dset_id,
                    xid="time",
                    yid="value",
                    count="count",
                ),
                observable=FitData(
                    self,
                    task=f"task_cap_{self.routes}_{self.dose}",
                    xid="time",
                    yid=self.yids[kr],
                ),
                metadata=CaptoprilMappingMetaData(
                    tissue=Tissue.PLASMA,
                    route=Route.PO,
                    dosing=Dosing.SINGLE,
                    health=Health.HEALTHY,
                    fasting=Fasting.FASTING,
                    data=PKPDData.PK
                ),
            )
        # console.print(mappings)
        return mappings

    def figures(self) -> Dict[str, Figure]:
        fig = Figure(
            experiment=self,
            sid="Figure 9",
            num_rows=1,
            num_cols=2,
            name=f"{self.__class__.__name__}",
            height=self.panel_height,
            width=self.panel_width * 2,
        )
        plots = fig.create_plots(
            xaxis=Axis(self.labels["time"], unit=self.units["time"]), legend=True
        )
        for ky, yid_unique in enumerate(["[Cve_cap]", "[Cve_captot]"]):
            kexp = 1
            plots[ky].set_yaxis(self.labels[yid_unique], unit=self.units[yid_unique])
            plots[ky].xaxis.max = 25
            plots[ky].xaxis.min = -0.5
            plots[ky].yaxis.max = 1.4
            # simulation
            plots[ky].add_data(
                task=f"task_cap_{self.routes}_{self.dose}",
                xid="time",
                yid=yid_unique,
                label=f"sim: {self.dose}mg {self.routes}",
                color=self.colors["control"],
            )
            for kr, dset_id in enumerate(self.clabels):
                yid = self.yids[kr]
                if yid == yid_unique:
                    plots[ky].add_data(
                        dataset=dset_id,
                        xid="time",
                        yid="value",
                        count="count",
                        label=f"exp{kexp}: {self.dose}mg {self.routes}" if ky==0 else f"exp: {self.dose}mg {self.routes}",
                        color=self.colors["control"],
                        marker=self.marker[f"exp{kexp}"]  if ky==0 else "s",
                    )
                    kexp += 1
        return {
            fig.sid: fig,
        }


if __name__ == "__main__":
    run_experiments(Kok1997, output_dir=Kok1997.__name__)
