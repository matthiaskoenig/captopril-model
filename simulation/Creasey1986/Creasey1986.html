"""Creasey1986"""

from typing import Dict

from sbmlsim.data import DataSet, load_pkdb_dataframe
from sbmlsim.fit import FitMapping, FitData
from sbmlutils.console import console

from pkdb_models.models.captopril.experiments.base_experiment import (
    CaptoprilSimulationExperiment,
)
from pkdb_models.models.captopril.experiments.metadata import Tissue, Route, Dosing, Health, Fasting, \
    CaptoprilMappingMetaData, PKPDData
from sbmlsim.plot import Axis, Figure
from sbmlsim.simulation import Timecourse, TimecourseSim

from pkdb_models.models.captopril.helpers import run_experiments

import numpy as np


class Creasey1986(CaptoprilSimulationExperiment):
    """Simulation experiment for Creasey1986.
    The pharmacokinetics of captopril were studied in an older healthy population.
    100 mg oral dose of captorpil was given with 250 mL of water.
    -fasting 8 hr before, 3 hours after
    """

    route = "PO"
    dose = 100  # mg
    yids = []
    clabels = []
    substances = []
    tissues = []

    colors = {
        "all": CaptoprilSimulationExperiment.subgroups_colors["all"],
    }
    marker = {
        "I1": "v",
        "I2": "o",
        "I3": "x",
        "I4": "^",
        "I5": "*",
        "I6": ">",
        "I7": "D",
        "all": "s"
    }

    def datasets(self) -> Dict[str, DataSet]:
        dsets = {}
        Q_ = self.Q_
        self.substances = []
        self.clabels = []
        self.yids = []
        self.tissues = []
        recovery_to_cumamount = Q_(self.dose, "mg") / Q_(100, "percent")
        for fig_id in ["Tab2", "Tab2Mean", "Tab3"]:
            df = load_pkdb_dataframe(f"{self.sid}_{fig_id}", data_path=self.data_path)
            for label, df_label in df.groupby("label"):
                dset = DataSet.from_df(df_label, self.ureg)
                if fig_id == "Tab3":
                    if "total captopril" in label:
                        dset.unit_conversion(
                            "mean", 1 / self.Mr.captot * recovery_to_cumamount
                        )
                        self.yids.append("Aurine_captot")
                        self.substances.append("captot")
                    else:
                        dset.unit_conversion(
                            "mean", 1 / self.Mr.cap * recovery_to_cumamount
                        )
                        self.yids.append("Aurine_cap")
                        self.substances.append("cap")
                    self.tissues.append("urine")
                else:
                    if "_I" in label:
                        a = "value"
                    else:
                        a = "mean"
                    if "total captopril" in label:
                        # converting from mg to mmole
                        dset.unit_conversion(a, 1 / self.Mr.captot)
                        self.yids.append("[Cve_captot]")
                        self.substances.append("captot")
                    elif "_captopril_" in label:
                        dset.unit_conversion(a, 1 / self.Mr.cap / self.BPRatios.cap)
                        self.yids.append("[Cve_cap]")
                        self.substances.append("cap")
                    else:
                        dset.unit_conversion(a, 1 / self.Mr.capme)
                        self.yids.append("[Cve_capme]")
                        self.substances.append("capme")
                    self.tissues.append("plasma")
                self.clabels.append(label)
                dsets[f"{label}"] = dset
        # console.print(dsets.keys())
        # console.print(dsets)
        return dsets

    def simulations(self) -> Dict[str, TimecourseSim]:
        dose = self.dose
        Q_ = self.Q_
        tcsims = {}
        tcsims[f"cap_{self.route}_{self.dose}"] = TimecourseSim(
            Timecourse(
                start=0,
                end=60 * 26,  # minutes
                steps=400,
                changes={
                    **self.default_changes(),
                    "BW": Q_(76.4, "kg"),
                    "HEIGHT": Q_(174, "cm"),
                    # "MAP": Q_((140+2*81)/3, "mmHg"),
                    f"SBP_ref": Q_(140, "mmHg"),
                    f"DBP_ref": Q_(81, "mmHg"),
                    "HCT": Q_(0.478, "dimensionless"),
                    f"{self.route}DOSE_cap": Q_(dose, "mg"),
                },
            )
        )
        return tcsims

    def fit_mappings(self) -> Dict[str, FitMapping]:
        mappings = {}
        for kr, dset_id in enumerate(self.clabels):
            substance = self.substances[kr]
            tissue = self.tissues[kr]
            subjects = dset_id.split("_")[-1]
            mappings[
                f"fm_cap{self.route}{self.dose}_{substance}_{tissue}_acute_fasting_{subjects}"
            ] = FitMapping(
                self,
                reference=FitData(
                    self,
                    dataset=dset_id,
                    xid="time",
                    yid="value" if "I" in subjects else "mean",
                    yid_sd=None if "I" in subjects else "mean_sd",
                    count="count",
                ),
                observable=FitData(
                    self,
                    task=f"task_cap_{self.route}_{self.dose}",
                    xid="time",
                    yid=self.yids[kr],
                ),
                metadata=CaptoprilMappingMetaData(
                    tissue=Tissue.PLASMA if "plasma" in tissue else Tissue.URINE,
                    route=Route.PO,
                    dosing=Dosing.SINGLE,
                    health=Health.HEALTHY,
                    fasting=Fasting.FASTING,
                    data=PKPDData.PK
                ),
            )
        # console.print(mappings)
        return mappings

    def figures(self) -> Dict[str, Figure]:
        fig1 = Figure(
            experiment=self,
            sid="Tabs 2",
            num_rows=1,
            num_cols=3,
            name=f"{self.__class__.__name__}",
            height=self.panel_height,
            width=self.panel_width * 3,
        )
        plots = fig1.create_plots(
            xaxis=Axis(self.labels["time"], unit=self.units["time"]), legend=True
        )

        for kp, yid_unique in enumerate(["[Cve_cap]", "[Cve_captot]", "[Cve_capme]"]):
            plots[kp].set_yaxis(
                self.labels[yid_unique], unit=self.units[yid_unique]
            )
            plots[kp].xaxis.max = 26
            # simulation
            plots[kp].add_data(
                task=f"task_cap_{self.route}_{self.dose}",
                xid="time",
                yid=yid_unique,
                label=f"sim all: {self.dose}mg PO",
                color=self.subgroups_colors["all"],
            )
            for kr, dset_id in enumerate(self.clabels):
                if yid_unique == self.yids[kr]:
                    plots[kp].add_data(
                        dataset=dset_id,
                        xid="time",
                        yid="value" if "I" in dset_id else "mean",
                        yid_sd=None if "I" in dset_id else "mean_sd",
                        count="count",
                        label=f"exp {dset_id.split("_")[-1]}: {self.dose}mg PO",
                        color=self.subgroups_colors["all"],
                        marker=self.marker[dset_id.split("_")[-1]],
                    )

        fig2 = Figure(
            experiment=self,
            sid="Tabs 3",
            num_rows=1,
            num_cols=2,
            name=f"{self.__class__.__name__}",
            height=self.panel_height,
            width=self.panel_width * 2,
        )
        plots = fig2.create_plots(
            xaxis=Axis(self.labels["time"], unit=self.units["time"]), legend=True
        )

        for kp, yid_unique in enumerate(
                ["Aurine_cap", "Aurine_captot"]):
            plots[kp].set_yaxis(
                self.labels[yid_unique], unit=self.units[yid_unique]
            )
            plots[kp].xaxis.max = 26
            # simulation
            plots[kp].add_data(
                task=f"task_cap_{self.route}_{self.dose}",
                xid="time",
                yid=yid_unique,
                label=f"sim all: {self.dose}mg PO",
                color=self.subgroups_colors["all"],
            )
            for kr, dset_id in enumerate(self.clabels):
                if yid_unique == self.yids[kr]:
                    plots[kp].add_data(
                        dataset=dset_id,
                        xid="time",
                        yid="value" if "I" in dset_id else "mean",
                        yid_sd=None if "I" in dset_id else "mean_sd",
                        count="count",
                        label=f"exp all: {self.dose}mg PO",
                        color=self.subgroups_colors["all"],
                    )

        return {
            fig1.sid: fig1,
            fig2.sid: fig2,
        }


if __name__ == "__main__":
    run_experiments(Creasey1986, output_dir=Creasey1986.__name__)
