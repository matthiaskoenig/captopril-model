"""Ohman1985"""
from copy import deepcopy
from typing import Dict

from sbmlsim.data import DataSet, load_pkdb_dataframe
from sbmlsim.fit import FitMapping, FitData
from sbmlutils.console import console

from pkdb_models.models.captopril.experiments.base_experiment import (
    CaptoprilSimulationExperiment,
)
from pkdb_models.models.captopril.experiments.metadata import Tissue, Route, Dosing, Health, Fasting, \
    CaptoprilMappingMetaData, PKPDData
from sbmlsim.plot import Axis, Figure
from sbmlsim.simulation import Timecourse, TimecourseSim

from pkdb_models.models.captopril.helpers import run_experiments


class Ohman1985(CaptoprilSimulationExperiment):
    """Simulation experiment for Ohman1985.
    The study describes PK of captopril in fed or fasted states, acute and chronic exposure
    """

    # important parameters from study
    route = "PO"
    dose = 25  # mg
    regimens = []
    states = []
    tissues = []

    colors = {
        "fasting": CaptoprilSimulationExperiment.fasting_colors["fasting"],
        "nonfasting": CaptoprilSimulationExperiment.fasting_colors["nonfasting"],
        "control": CaptoprilSimulationExperiment.renal_colors["Control"]
    }

    element_unique = ["MAP", "total captopril", "_captopril_urine", "captopril"]

    # creating dataset from PK data
    def datasets(self) -> Dict[str, DataSet]:
        Q_ = self.Q_

        dsets = {}
        self.reset_state()
        self.states = []
        self.tissues = []
        self.regimens = []
        for fig_id in ["Fig1", "Fig2", "Fig3", "Tab1"]:
            df = load_pkdb_dataframe(f"{self.sid}_{fig_id}", data_path=self.data_path)
            for label, df_label in df.groupby("label"):
                dset = DataSet.from_df(df_label, self.ureg)
                for element in self.element_unique:
                    if element in label and "excretion" not in label:
                        self.data_collection(element=element, label=label, dose=self.dose)

                        if "recovery" in label:
                            self.tissues.append("urine")
                            dset.unit_conversion(
                                self.data_type, self.conversion_factor
                            )
                        elif "plasma" in label:
                            self.tissues.append("plasma")
                        else:
                            self.tissues.append("nr")
                        if "nonfasting" in label:
                            self.states.append("nonfasting")
                        else:
                            self.states.append("fasting")
                        if "chronic" in label:
                            self.regimens.append("chronic")
                        else:
                            self.regimens.append("acute")

                        if not "nonfasting" in label and "chronic" in label:
                            dset["time"] = dset["time"] - 696
                        if "nonfasting" in label and "chronic" in label:
                            dset["time"] = dset["time"] - 864

                        dsets[f"{label}"] = dset

                        break
        # console.print(dsets.keys())
        # console.print(dsets)
        return dsets

    def simulations(self) -> Dict[str, TimecourseSim]:
        Q_ = self.Q_
        tcsims = {}

        tc_start = Timecourse(
            start=0,
            end=24 * 60,  # minutes
            steps=400,
            changes={
                **self.default_changes(),
                "BW": Q_(73, "kg"),
                f"{self.route}DOSE_cap": Q_(self.dose, "mg"),
                "GU__F_cap_abs": Q_(self.fasting_map["fasted"], "dimensionless")
            },
        )
        tc_end = Timecourse(
            start=0,
            end=1 * 60,  # minutes
            steps=300,
            changes={
                f"{self.route}DOSE_cap": Q_(0, "mg"),
            },
        )
        tc = Timecourse(
            start=0,
            end=6 * 60,  # minutes
            steps=300,
            changes={
                f"{self.route}DOSE_cap": Q_(self.dose, "mg"),
            },
        )
        tc_night = Timecourse(
            start=0,
            end=6 * 60,  # minutes
            steps=300,
            changes={
                f"{self.route}DOSE_cap": Q_(0, "mg"),
            },
        )
        tc_fasting_end = Timecourse(
            start=0,
            end=24 * 60,  # minutes
            steps=400,
            changes={
                f"{self.route}DOSE_cap": Q_(self.dose, "mg"),
                "GU__F_cap_abs": Q_(self.fasting_map["fasted"], "dimensionless")
            },
        )
        tc_nonfasting_end = Timecourse(
            start=0,
            end=25 * 60,  # minutes
            steps=400,
            changes={
                f"{self.route}DOSE_cap": Q_(self.dose, "mg"),
                "GU__F_cap_abs": Q_(self.fasting_map["nonfasting"], "dimensionless")
            },
        )

        tcsims[f"cap_{self.route}_{self.dose}_acute_fasting"] = TimecourseSim(
            timecourses=[tc_start, tc_end],
        )
        tcsims[f"cap_{self.route}_{self.dose}_chronic_fasting"] = TimecourseSim(
            timecourses=[tc_start] + [tc, tc, tc, tc_night] * (4 * 7) + [tc_fasting_end, tc_end],
            time_offset=-(24 + 4 * 7 * 24) * 60
        )
        tcsims[f"cap_{self.route}_{self.dose}_chronic_nonfasting"] = TimecourseSim(
            timecourses=[tc_start] + ([tc, tc, tc, tc_night]) * (4 * 7) + [tc_fasting_end] +
                        ([tc, tc, tc, tc_night]) * (1 * 6) + [tc_nonfasting_end],
            time_offset=-(24 + 4 * 7 * 24 + 24 + 1 * 6 * 24) * 60
        )

        return tcsims

    def fit_mappings(self) -> Dict[str, FitMapping]:
        mappings = {}
        for kr, label in enumerate(self.clabels):
            element = self.element_ids[kr]
            regimen = self.regimens[kr]
            state = self.states[kr]
            tissue = self.tissues[kr]
            mappings[
                f"fm_cap{self.route}{self.dose}_{element}_{tissue}_{regimen}_{state}"
            ] = FitMapping(
                self,
                reference=FitData(
                    self,
                    dataset=label,
                    xid="time",
                    yid="mean",
                    yid_sd="mean_sd",
                    count="count",
                ),
                observable=FitData(
                    self,
                    task=f"task_cap_{self.route}_{self.dose}_{regimen}_{state}",
                    xid="time",
                    yid=self.yids[kr],
                ),
                metadata=CaptoprilMappingMetaData(
                    tissue=Tissue.PLASMA if tissue == "plasma" else Tissue.URINE,
                    route=Route.PO,
                    dosing=Dosing.SINGLE if regimen == "acute" else Dosing.MULTIPLE,
                    health=Health.HEALTHY,
                    fasting=Fasting.NONFASTING if state == "nonfasting" else Fasting.FASTING,
                    data=PKPDData.PD if element == "MAP" else PKPDData.PK,
                ),
            )
        return mappings

    def figures(self) -> Dict[str, Figure]:
        return {
            **self.pk_figures(),
            #**self.pd_figures()
        }

    def pk_figures(self) -> Dict[str, Figure]:
        fig1 = Figure(
            experiment=self,
            sid="PK plasma",
            num_rows=2,
            num_cols=2,
            name=f"{self.__class__.__name__}",
            height=self.panel_height * 2,
            width=self.panel_width * 2,
        )
        plots = fig1.create_plots(
            xaxis=Axis(self.labels["time"], unit="day"), legend=True
        )

        for ky, yid_unique in enumerate(["[Cve_cap]", "[Cve_captot]"]):
            for state, regimen, plot_indexes, xlims in zip(["fasting", "fasting", "nonfasting"], ["acute", "chronic", "chronic"], [[0, 1], [2, 3], [2, 3]], [[-0.1, 1.1], [-0.6, 1.1], [-0.6, 1.1]]):
                if yid_unique == "[Cve_cap]":
                    plot_index = plot_indexes[0]
                else:
                    plot_index = plot_indexes[1]
                plots[plot_index].set_yaxis(self.labels[yid_unique], unit=self.units[yid_unique])
                plots[plot_index].xaxis.min = xlims[0]
                plots[plot_index].xaxis.max = xlims[1]
                plots[plot_index].yaxis.max = 7.0
                # simulation
                plots[plot_index].add_data(
                    task=f"task_cap_{self.route}_{self.dose}_{regimen}_{state}",
                    xid="time",
                    yid=yid_unique,
                    label=f"{state} sim: {regimen} {self.dose}mg PO",
                    color=self.fasting_colors[state],
                )
                for ky, yid in enumerate(self.yids):
                    if yid == yid_unique and regimen == self.regimens[ky] and state == self.states[ky]:
                        plots[plot_index].add_data(
                            dataset=self.clabels[ky],
                            xid="time",
                            yid="mean",
                            #yid_sd="mean_sd",
                            count="count",
                            label=f"{self.states[ky]} exp: {self.regimens[ky]} {self.dose}mg PO",
                            color=self.fasting_colors[state],
                        )

        fig2 = Figure(
            experiment=self,
            sid="PK urine",
            num_rows=1,
            num_cols=1,
            name=f"{self.__class__.__name__}",
            height=self.panel_height,
            width=self.panel_width,
        )
        plots = fig2.create_plots(
            xaxis=Axis(self.labels["time"], unit="day"), legend=True
        )
        plots[0].set_yaxis(self.labels["Aurine_cap"], unit=self.units["Aurine_cap"])
        # simulation
        plots[0].add_data(
            task=f"task_cap_{self.route}_{self.dose}_acute_fasting",
            xid="time",
            yid="Aurine_cap",
            label=f"sim: {self.dose}mg PO",
            color=self.fasting_colors["fasting"],
        )
        for ky, yid in enumerate(self.yids):
            if yid == "Aurine_cap":
                plots[0].add_data(
                    dataset=self.clabels[ky],
                    xid="time",
                    yid="mean",
                    #yid_sd="mean_sd",
                    count="count",
                    label=f"{self.states[ky]} exp: {self.regimens[ky]} {self.dose}mg PO",
                    color=self.fasting_colors["fasting"],
                )

        return {
            fig1.sid: fig1,
            fig2.sid: fig2,
        }

    def pd_figures(self) -> Dict[str, Figure]:
        fig = Figure(
            experiment=self,
            sid="PD",
            num_rows=1,
            num_cols=1,
            name=f"{self.__class__.__name__}",
        )
        plots = fig.create_plots(
            xaxis=Axis(self.labels["time"], unit="day"), legend=True
        )
        for kyu, yid_unique in enumerate(["MAP"]):
            plots[kyu].set_yaxis(self.labels[yid_unique], unit=self.units[yid_unique])
            # simulation
            plots[kyu].add_data(
                task=f"task_cap_{self.route}_{self.dose}",
                xid="time",
                yid=yid_unique,
                label=f"fasting sim: {self.dose}mg PO",
                color=self.fasting_colors[f"fasting"],
            )
            for ky, yid in enumerate(self.yids):
                if yid == yid_unique:
                    plots[kyu].add_data(
                        dataset=self.clabels[ky],
                        xid="time",
                        yid="mean",
                        #yid_sd="mean_sd",
                        count="count",
                        label=f"{self.regimens[ky]} {self.states[ky]} exp: {self.dose}mg PO",
                        color=self.fasting_colors[f"{self.states[ky]}"],
                    )

        return {
            fig.sid: fig,
        }
