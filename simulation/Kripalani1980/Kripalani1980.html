"""Kripalani1980"""

from typing import Dict

from sbmlsim.data import DataSet, load_pkdb_dataframe
from sbmlsim.fit import FitMapping, FitData
from sbmlutils.console import console

from pkdb_models.models.captopril.experiments.base_experiment import (
    CaptoprilSimulationExperiment,
)
from pkdb_models.models.captopril.experiments.metadata import Tissue, Route, Dosing, Health, Fasting, \
    CaptoprilMappingMetaData, PKPDData
from sbmlsim.plot import Axis, Figure
from sbmlsim.simulation import Timecourse, TimecourseSim

from pkdb_models.models.captopril.helpers import run_experiments


class Kripalani1980(CaptoprilSimulationExperiment):
    """Simulation experiment for Kripalani1980.
    Captopril pharmacokinetics in healthy subjects.
    - fasted for at least 8 hr before and 3hr after drug
    """

    route = "PO"
    dose = 1.37  # mg/kg
    yids = []
    clabels = []
    tissues = []
    substances = []

    colors = {"control": CaptoprilSimulationExperiment.renal_colors["Control"]}

    def datasets(self) -> Dict[str, DataSet]:
        dsets = {}
        Q_ = self.Q_
        self.clabels = []
        self.tissues = []
        self.yids = []
        self.substances = []
        recovery_to_cumamount = Q_(self.dose, "mg/kg") / Q_(100, "percent") * Q_(73.3, "kg")
        for fig_id in ["Fig2", "Tab3", "Tab1"]:
            df = load_pkdb_dataframe(f"{self.sid}_{fig_id}", data_path=self.data_path)
            for label, df_label in df.groupby("label"):
                dset = DataSet.from_df(df_label, self.ureg)
                if fig_id == "Fig2":
                    if "captopril+" in label:
                        dset.unit_conversion(
                            "mean", 1 / self.Mr.capall / self.BPRatios.cap
                        )
                        self.yids.append("[Cve_capall]")
                        self.tissues.append("plasma")
                        self.substances.append("capall")
                    elif "_captopril_" in label:
                        dset.unit_conversion(
                            "mean", 1 / self.Mr.cap / self.BPRatios.cap
                        )
                        self.yids.append("[Cve_cap]")
                        self.tissues.append("plasma")
                        self.substances.append("cap")
                    else:
                        continue
                if fig_id == "Tab3":
                    if "captopril disulfide metabolites" in label:
                        dset.unit_conversion(
                            "mean", 1 / self.Mr.capss * recovery_to_cumamount
                        )
                        self.yids.append("Aurine_capss")
                        self.tissues.append("urine")
                        self.substances.append("capss")
                    elif "_captopril_" in label:
                        dset.unit_conversion(
                            "mean", 1 / self.Mr.cap * recovery_to_cumamount
                        )
                        self.yids.append("Aurine_cap")
                        self.tissues.append("urine")
                        self.substances.append("cap")
                    else:
                        continue
                if fig_id == "Tab1":

                    if "urine" in label:
                        self.yids.append("Aurine_capall")
                        self.tissues.append("urine")
                        self.substances.append("capall")
                        dset.unit_conversion(
                            "mean", 1 / self.Mr.capall * recovery_to_cumamount
                        )
                    else:
                        self.yids.append("Afeces_captot")
                        self.tissues.append("feces")
                        self.substances.append("captot")
                        dset.unit_conversion(
                            "mean", 1 / self.Mr.captot * recovery_to_cumamount
                        )
                self.clabels.append(label)
                dsets[f"{label}"] = dset
        # console.print(dsets.keys())
        # console.print(dsets)
        # console.print(self.clabels)
        # console.print(self.yids)
        return dsets

    def simulations(self) -> Dict[str, TimecourseSim]:
        # for what?
        Q_ = self.Q_
        tcsims = {}
        tcsims[f"cap_{self.route}_{self.dose}"] = TimecourseSim(
            Timecourse(
                start=0,
                end=121 * 60,  # minutes
                steps=400,
                changes={
                    **self.default_changes(),
                    "BW": Q_(73.3, "kg"),
                    "HEIGHT": Q_(178.3, "cm"),
                    f"{self.route}DOSE_cap": Q_(self.dose, "mg/kg") * Q_(73.3, "kg"),
                },
            )
        )
        return tcsims

    def fit_mappings(self) -> Dict[str, FitMapping]:
        mappings = {}
        for kr, dset_id in enumerate(self.clabels):
            tissue = self.tissues[kr]
            substance = self.substances[kr]
            mappings[
                f"fm_cap{self.route}{self.dose}_{substance}_{tissue}_acute_fasting_all"
            ] = FitMapping(
                self,
                reference=FitData(
                    self,
                    dataset=dset_id,
                    xid="time",
                    yid="mean",
                    yid_sd="mean_sd",
                    count="count",
                ),
                observable=FitData(
                    self,
                    task=f"task_cap_{self.route}_{self.dose}",
                    xid="time",
                    yid=self.yids[kr],
                ),
                metadata=CaptoprilMappingMetaData(
                    tissue=(
                        Tissue.PLASMA
                        if "blood" in dset_id
                        else Tissue.URINE if "urine" in dset_id else Tissue.FECES
                    ),
                    route=Route.PO,
                    dosing=Dosing.SINGLE,
                    health=Health.HEALTHY,
                    fasting=Fasting.FASTING,
                    data=PKPDData.PK
                ),
            )
        return mappings

    def figures(self) -> Dict[str, Figure]:
        fig1 = Figure(
            experiment=self,
            sid="Figure 2",
            num_rows=1,
            num_cols=2,
            name=f"{self.__class__.__name__}",
            height=self.panel_height,
            width=self.panel_width * 2,
        )
        plots = fig1.create_plots(
            xaxis=Axis(self.labels["time"], unit=self.units["time"]), legend=True
        )

        for ky, yid_unique in enumerate(["[Cve_cap]", "[Cve_capall]"]):
            plots[ky].set_yaxis(
                self.labels[yid_unique], unit=self.units[yid_unique]
            )
            plots[ky].xaxis.min = -0.01
            plots[ky].xaxis.max = 15
            plots[ky].yaxis.max = 11

            for kr, dset_id in enumerate(self.clabels):
                yid = self.yids[kr]
                if yid == yid_unique:
                    # dataset
                    plots[ky].add_data(
                        dataset=dset_id,
                        xid="time",
                        yid="mean",
                        yid_sd="mean_sd",
                        count="count",
                        label=f"exp: {self.dose}mg/kg {self.route}",
                        color=self.colors["control"],
                    )
            # simulation
            plots[ky].add_data(
                task=f"task_cap_{self.route}_{self.dose}",
                xid="time",
                yid=yid_unique,
                label=f"sim: {self.dose}mg/kg {self.route}",
                color=self.colors["control"],
            )

        fig2 = Figure(
            experiment=self,
            sid="Tabs 1,3",
            num_rows=1,
            num_cols=4,
            name=f"{self.__class__.__name__}",
            height=self.panel_height,
            width=self.panel_width * 4,
        )
        plots = fig2.create_plots(
            xaxis=Axis(self.labels["time"], unit=self.units["time"]), legend=True
        )
        for ky, yid_unique in enumerate(["Aurine_cap", "Aurine_capall", "Aurine_capss", "Afeces_captot"]):
            plots[ky].set_yaxis(
                self.labels[yid_unique], unit=self.units[yid_unique]
            )

            for kr, dset_id in enumerate(self.clabels):
                yid = self.yids[kr]
                if yid == yid_unique:
                    # dataset
                    plots[ky].add_data(
                        dataset=dset_id,
                        xid="time",
                        yid="mean",
                        yid_sd="mean_sd",
                        count="count",
                        label=f"exp: {self.dose}mg/kg {self.route}",
                        color=self.colors["control"],
                    )
            # simulation
            plots[ky].add_data(
                task=f"task_cap_{self.route}_{self.dose}",
                xid="time",
                yid=yid_unique,
                label=f"sim: {self.dose}mg/kg {self.route}",
                color=self.colors["control"],
            )

        return {
            fig1.sid: fig1,
            fig2.sid: fig2,
        }


if __name__ == "__main__":
    run_experiments(Kripalani1980, output_dir=Kripalani1980.__name__)
