"""Creasey1988."""

from typing import Dict

from sbmlsim.data import DataSet, load_pkdb_dataframe
from sbmlsim.fit import FitMapping, FitData
from sbmlutils.console import console

from pkdb_models.models.captopril.experiments.base_experiment import (
    CaptoprilSimulationExperiment,
)
from pkdb_models.models.captopril.experiments.metadata import Tissue, Route, Dosing, Health, Fasting, \
    CaptoprilMappingMetaData, PKPDData
from sbmlsim.plot import Axis, Figure
from sbmlsim.simulation import Timecourse, TimecourseSim

from pkdb_models.models.captopril.helpers import run_experiments


class Creasey1988(CaptoprilSimulationExperiment):
    """Simulation experiment for Creasey1988.
    The pharmacokinetic characteristics of intravenously-administered captopril were investigated in healthy men.
    The subjects were fasted for 8 hr to and 2 hr after drug administration
    """

    route = "IV"
    doses = []  # mg
    yids = []
    clabels = []
    substances = []

    colors = {
        "11.4": CaptoprilSimulationExperiment.doses_colors["11.4mg"],
        "2.78": CaptoprilSimulationExperiment.doses_colors["2.78mg"],
        "5.67": CaptoprilSimulationExperiment.doses_colors["5.67mg"],
    }

    def datasets(self) -> Dict[str, DataSet]:
        dsets = {}
        self.clabels = []
        self.doses = []
        self.yids = []
        self.substances = []
        for fig_id in ["Fig1"]:
            df = load_pkdb_dataframe(f"{self.sid}_{fig_id}", data_path=self.data_path)
            for label, df_label in df.groupby("label"):
                dset = DataSet.from_df(df_label, self.ureg)
                self.clabels.append(label)
                if "2.78" in label:
                    self.doses.append(2.78)
                elif "5.67" in label:
                    self.doses.append(5.67)
                elif "11.4" in label:
                    self.doses.append(11.4)
                if "_captopril_" in label:
                    dset.unit_conversion("mean", 1 / self.Mr.cap / self.BPRatios.cap)
                    self.yids.append("[Cve_cap]")
                    self.substances.append("cap")
                if "_captopril+" in label:
                    dset.unit_conversion("mean", 1 / self.Mr.capall / self.BPRatios.cap)
                    self.yids.append("[Cve_capall]")
                    self.substances.append("capall")
                dsets[f"{label}"] = dset
        # console.print(dsets.keys())
        # console.print(dsets)
        return dsets

    def simulations(self) -> Dict[str, TimecourseSim]:
        Q_ = self.Q_
        tcsims = {}
        for k, dose in enumerate(self.doses):
            tcsims[f"cap_{self.route}_{dose}"] = TimecourseSim(
                Timecourse(
                    start=0,
                    end=13 * 60,  # minutes
                    steps=400,
                    changes={
                        **self.default_changes(),
                        f"{self.route}DOSE_cap": Q_(dose, "mg"),
                    },
                )
            )
        return tcsims

    def fit_mappings(self) -> Dict[str, FitMapping]:
        mappings = {}
        for kr, dose in enumerate(self.doses):
            substance = self.substances[kr]
            subjects = self.clabels[kr].split("_")[-1]
            mappings[
                f"fm_cap{self.route}{dose}_{substance}_plasma_acute_fasting_{subjects}"
            ] = FitMapping(
                self,
                reference=FitData(
                    self,
                    dataset=self.clabels[kr],
                    xid="time",
                    yid="mean",
                    yid_sd="mean_sd",
                    count="count",
                ),
                observable=FitData(
                    self,
                    task=f"task_cap_{self.route}_{dose}",
                    xid="time",
                    yid=self.yids[kr],
                ),
                metadata=CaptoprilMappingMetaData(
                    tissue=Tissue.PLASMA,
                    route=Route.IV,
                    dosing=Dosing.SINGLE,
                    health=Health.HEALTHY,
                    fasting=Fasting.FASTING,
                    data=PKPDData.PK
                ),
            )
        # console.print(mappings)
        return mappings

    def figures(self) -> Dict[str, Figure]:
        fig = Figure(
            experiment=self,
            sid="Figure 1",
            num_rows=1,
            num_cols=2,
            name=f"{self.__class__.__name__}",
            height=self.panel_height,
            width=self.panel_width * 2,
        )
        plots = fig.create_plots(
            xaxis=Axis(self.labels["time"], unit=self.units["time"]), legend=True
        )
        for kr, dose in sorted(enumerate(self.doses), reverse=True, key=lambda x: x[1]):
            data = self.clabels[kr]
            if "_captopril_" in data:
                a = 0
            else:
                a = 1
            plots[a].set_yaxis(
                self.labels[self.yids[kr]], unit=self.units[self.yids[kr]]
            )
            # simulation
            plots[a].add_data(
                task=f"task_cap_{self.route}_{dose}",
                xid="time",
                yid=self.yids[kr],
                label=f"sim: {dose}mg IV",
                color=self.colors[f"{dose}"],
            )
            # dataset
            plots[a].add_data(
                dataset=data,
                xid="time",
                yid="mean",
                yid_sd="mean_sd",
                count="count",
                label=f"exp: {dose}mg IV",
                color=self.colors[f"{dose}"],
            )
        return {
            fig.sid: fig,
        }


if __name__ == "__main__":
    run_experiments(Creasey1988, output_dir=Creasey1988.__name__)
