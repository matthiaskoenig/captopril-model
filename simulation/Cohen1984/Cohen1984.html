"""Cohen1984"""

from typing import Dict, Tuple, List, Hashable

import numpy as np
from sbmlsim.data import DataSet, load_pkdb_dataframe
from sbmlsim.fit import FitMapping, FitData
from sbmlutils.console import console

from pkdb_models.models.captopril.experiments.base_experiment import (
    CaptoprilSimulationExperiment,
)
from pkdb_models.models.captopril.experiments.metadata import Tissue, Route, Dosing, Health, Fasting, \
    CaptoprilMappingMetaData, PKPDData

from sbmlsim.plot import Axis, Figure
from sbmlsim.simulation import Timecourse, TimecourseSim

from pkdb_models.models.captopril.helpers import run_experiments


class Cohen1984(CaptoprilSimulationExperiment):
    """Simulation experiment for Cohen1984.
    Gas-chromatography-selected-ion monitoring mass sepectrometry-determination of
    the S-methyl metabolite of captopril in human plasma after oral administration.
    Also, total and unchanged captopril was determined and captopril disulfide was calculated.
    - no information about food
    """

    route = "PO"
    dose = 100
    yids = []
    substances = []

    marker = {
        "I1": "v",
        "I2": "o",
        "I3": "x",
        "I4": "^",
        "I5": "*",
        "I6": ">",
        "I7": "D",
    }

    colors = {"control": CaptoprilSimulationExperiment.renal_colors["Control"]}

    clabels = []
    substances = []

    def datasets(self) -> dict[str, DataSet]:
        dsets = {}
        self.substances = []
        self.clabels = []
        self.yids = []
        for fig_id in ["Tab4Mean", "Tab3"]:
            df = load_pkdb_dataframe(f"{self.sid}_{fig_id}", data_path=self.data_path)
            for label, df_label in df.groupby("label"):
                dset = DataSet.from_df(df_label, self.ureg)
                if "_I" in label:
                    a = "value"
                else:
                    a = "mean"
                if "total captopril" in label:
                    dset.unit_conversion(a, 1 / self.Mr.captot)
                    self.substances.append("captot")
                    self.yids.append("[Cve_captot]")
                elif "captopril disulfide metabolites" in label:
                    dset.unit_conversion(a, 1 / self.Mr.capss)
                    self.substances.append("capss")
                    self.yids.append("[Cve_capss]")
                elif "concentration_captopril_all" in label:
                    dset.unit_conversion(a, 1 / self.Mr.cap)
                    self.substances.append("cap")
                    self.yids.append("[Cve_cap]")
                elif "_S-" in label:
                    dset.unit_conversion(a, 1 / self.Mr.capme)
                    self.substances.append("capme")
                    self.yids.append("[Cve_capme]")
                self.clabels.append(label)
                dsets[f"{label}"] = dset
        # console.print(dsets.keys())
        # console.print(dsets)
        return dsets

    def simulations(self) -> Dict[str, TimecourseSim]:
        Q_ = self.Q_
        tcsims = {}
        # single dose
        tcsims[f"cap_{self.route}_{self.dose}"] = TimecourseSim(
            Timecourse(
                start=0,
                end=26 * 60,  # minutes
                steps=400,
                changes={
                    **self.default_changes(),
                    f"{self.route.upper()}DOSE_cap": Q_(self.dose, "mg"),
                },
            )
        )
        return tcsims

    def fit_mappings(self) -> Dict[str, FitMapping]:
        mappings = {}
        for k, dset_id in enumerate(self.clabels):
            substance = self.substances[k]
            subjects = dset_id.split("_")[-1]
            mappings[
                f"fm_cap{self.route}{self.dose}_{substance}_plasma_acute_nrfasting_{subjects}"
            ] = FitMapping(
                self,
                reference=FitData(
                    self,
                    dataset=dset_id,
                    xid="time",
                    yid="mean" if "all" in dset_id else "value",
                    yid_sd="mean_sd" if "all" in dset_id else None,
                    count="count",
                ),
                observable=FitData(
                    self,
                    task=f"task_cap_{self.route}_{self.dose}",
                    xid="time",
                    yid=self.yids[k],
                ),
                metadata=CaptoprilMappingMetaData(
                    tissue=Tissue.PLASMA,
                    route=Route.PO,
                    dosing=Dosing.SINGLE,
                    health=Health.HEALTHY,
                    fasting=Fasting.NR,
                    data=PKPDData.PK
                ),
            )
        # console.print(mappings)
        return mappings

    def figures(self) -> Dict[str, Figure]:
        fig = Figure(
            experiment=self,
            sid="Tabs 3,4",
            num_rows=1,
            num_cols=4,
            name=f"{self.__class__.__name__}",
            height=self.panel_height,
            width=self.panel_width * 4,
        )
        plots = fig.create_plots(
            xaxis=Axis(self.labels["time"], unit=self.units["time"]), legend=True
        )
        for k, dset_id in enumerate(self.clabels):
            if "_captopril_" in dset_id:
                a = 0
            elif "total captopril" in dset_id:
                a = 1
            elif "captopril disulfide metabolites" in dset_id:
                a = 2
            else:
                a = 3
            plots[a].set_yaxis(self.labels[self.yids[k]], unit=self.units[self.yids[k]])
            if k < 4:
                # simulation
                plots[a].add_data(
                    task=f"task_cap_{self.route}_{self.dose}",
                    xid="time",
                    yid=self.yids[k],
                    label=f"sim all: {self.dose}mg PO",
                    color=self.colors["control"],
                )
            plots[a].add_data(
                dataset=dset_id,
                xid="time",
                yid="mean" if "all" in dset_id else "value",
                yid_sd="mean_sd" if "all" in dset_id else None,
                count="count",
                label=f"exp {dset_id.split("_")[-1]}: {self.dose}mg PO",
                color=self.colors["control"],
                marker=self.marker[dset_id.split("_")[-1]] if "all" not in dset_id else "s",
            )
        return {
            fig.sid: fig,
        }


if __name__ == "__main__":
    run_experiments(Cohen1984, output_dir=Cohen1984.__name__)
